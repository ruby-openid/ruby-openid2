def initialize(text)
  super
  temp_arr = text.scan(/^<!--\s*(.*?)\s*-->$/m)
  raise HTMLTokenizerError, "Text passed to HTMLComment.initialize is not a comment" if temp_arr[0].nil?

  @contents = temp_arr[0][0]
end
